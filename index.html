12Анализ программы с условием

Ниже приведена программа, записанная на четырёх языках программирования. Было проведено 9 запусков программы, при которых в качестве значений переменных вводились пары чисел. Сколько было запусков, при которых программа напечатала «ДА»?

Алгоритмический язык

алг
нач
  цел s, k
  ввод s
  ввод k
  если s <= 2 и k <= 3
    то вывод "ДА"
    иначе вывод "НЕТ"
  все
кон

Python

s = int(input())
k = int(input())
if s <= 2 and k <= 3:
    print("ДА")
else:
    print("НЕТ")

C++

#include <iostream>
using namespace std;

int main() {
  int s, k;
  cin >> s;
  cin >> k;
  if (s <= 2 && k <= 3)
    cout << "ДА";
  else
    cout << "НЕТ";
  return 0;
}

Паскаль

var s, k: integer;
begin
  readln(s);
  readln(k);
  if (s <= 2) and (k <= 3)
    then writeln('ДА')
    else writeln('НЕТ')
end.

Python

s = 0
for i in range(10):
    if i % 4 == 0:
        s = s + 1
print(s)

C++

#include <iostream>
using namespace std;

int main() {
  int s, i;
  s = 0;
  for (i = 0; i < 10; i++)
    if (i % 4 == 0)
      s = s + 1;
  cout << s;
  return 0;
}
